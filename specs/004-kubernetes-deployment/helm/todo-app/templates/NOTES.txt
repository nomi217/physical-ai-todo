ğŸš€ **Getting Started:**

1. Check the status of your pods:
   kubectl get pods -l app.kubernetes.io/instance={{ .Release.Name }}

2. Watch pods until they're running:
   kubectl get pods -w

3. Get the application URL:
{{- if .Values.frontend.enabled }}
  {{- if eq .Values.frontend.service.type "LoadBalancer" }}
   export SERVICE_IP=$(kubectl get svc {{ include "todo-app.fullname" . }}-frontend -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
   echo "Frontend URL: http://$SERVICE_IP:{{ .Values.frontend.service.port }}"
  {{- else if eq .Values.frontend.service.type "NodePort" }}
   export NODE_PORT=$(kubectl get svc {{ include "todo-app.fullname" . }}-frontend -o jsonpath='{.spec.ports[0].nodePort}')
   export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
   echo "Frontend URL: http://$NODE_IP:$NODE_PORT"
  {{- else }}
   kubectl port-forward svc/{{ include "todo-app.fullname" . }}-frontend {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }}
   echo "Frontend URL: http://localhost:{{ .Values.frontend.service.port }}"
  {{- end }}
{{- end }}

4. Access backend API docs:
{{- if .Values.backend.enabled }}
   kubectl port-forward svc/{{ include "todo-app.fullname" . }}-backend {{ .Values.backend.service.port }}:{{ .Values.backend.service.port }}
   echo "API Docs: http://localhost:{{ .Values.backend.service.port }}/docs"
